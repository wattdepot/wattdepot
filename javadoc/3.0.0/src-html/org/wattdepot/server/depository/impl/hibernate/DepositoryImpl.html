019package org.wattdepot.server.depository.impl.hibernate;
020
021import javax.persistence.Entity;
022import javax.persistence.GeneratedValue;
023import javax.persistence.Id;
024import javax.persistence.ManyToOne;
025import javax.persistence.Table;
026import javax.persistence.Transient;
027
028import org.wattdepot.common.domainmodel.Depository;
029import org.wattdepot.server.ServerProperties;
030
031/**
032 * DepositoryImpl - Implementation of Depository that stores the Measurements in
033 * a Hibernate database.
034 * 
035 * @author Cam Moore
036 * 
037 */
038@Entity
039@Table(name = "DEPOSITORIES")
040public class DepositoryImpl {
041
042  /** Database primary key. */
043  @Id
044  @GeneratedValue
045  private Long pk;
046  /** The unique id it is also a slug used in URIs. */
047  private String id;
048  /** Name of the Depository. */
049  private String name;
050  /** MeasurementType of the measurements stored in this depository. */
051  @ManyToOne
052  private MeasurementTypeImpl type;
053  /** This depository's organization. */
054  @ManyToOne
055  private OrganizationImpl org;
056  @Transient
057  private ServerProperties serverProperties;
058
059  /**
060   * Default constructor.
061   */
062  public DepositoryImpl() {
063    super();
064  }
065
066  /**
067   * @param id the id.
068   * @param name the name.
069   * @param type the type.
070   * @param org the organization.
071   */
072  public DepositoryImpl(String id, String name, MeasurementTypeImpl type, OrganizationImpl org) {
073    this.id = id;
074    this.name = name;
075    this.type = type;
076    this.org = org;
077  }
078
079  /*
080   * (non-Javadoc)
081   * 
082   * @see java.lang.Object#equals(java.lang.Object)
083   */
084  @Override
085  public boolean equals(Object obj) {
086    if (this == obj) {
087      return true;
088    }
089    if (obj == null) {
090      return false;
091    }
092    if (getClass() != obj.getClass()) {
093      return false;
094    }
095    DepositoryImpl other = (DepositoryImpl) obj;
096    if (id == null) {
097      if (other.id != null) {
098        return false;
099      }
100    }
101    else if (!id.equals(other.id)) {
102      return false;
103    }
104    if (name == null) {
105      if (other.name != null) {
106        return false;
107      }
108    }
109    else if (!name.equals(other.name)) {
110      return false;
111    }
112    if (org == null) {
113      if (other.org != null) {
114        return false;
115      }
116    }
117    else if (!org.equals(other.org)) {
118      return false;
119    }
120    if (pk == null) {
121      if (other.pk != null) {
122        return false;
123      }
124    }
125    else if (!pk.equals(other.pk)) {
126      return false;
127    }
128    if (type == null) {
129      if (other.type != null) {
130        return false;
131      }
132    }
133    else if (!type.equals(other.type)) {
134      return false;
135    }
136    return true;
137  }
138
139
140  /**
141   * @return the id
142   */
143  public String getId() {
144    return id;
145  }
146
147  /**
148   * @return the name
149   */
150  public String getName() {
151    return name;
152  }
153
154  /**
155   * @return the org
156   */
157  public OrganizationImpl getOrg() {
158    return org;
159  }
160
161  /**
162   * @return the pk
163   */
164  public Long getPk() {
165    return pk;
166  }
167
168  /**
169   * @return the type
170   */
171  public MeasurementTypeImpl getType() {
172    return type;
173  }
174
175  /*
176   * (non-Javadoc)
177   * 
178   * @see java.lang.Object#hashCode()
179   */
180  @Override
181  public int hashCode() {
182    final int prime = 31;
183    int result = 1;
184    result = prime * result + ((id == null) ? 0 : id.hashCode());
185    result = prime * result + ((name == null) ? 0 : name.hashCode());
186    result = prime * result + ((org == null) ? 0 : org.hashCode());
187    result = prime * result + ((pk == null) ? 0 : pk.hashCode());
188    result = prime * result + ((type == null) ? 0 : type.hashCode());
189    return result;
190  }
191
192  /**
193   * @return the serverProperties
194   */
195  public ServerProperties serverProperties() {
196    return serverProperties;
197  }
198
199  /**
200   * @param serverProperties the serverProperties to set
201   */
202  public void serverProperties(ServerProperties serverProperties) {
203    this.serverProperties = serverProperties;
204  }
205
206  /**
207   * @param id the id to set
208   */
209  public void setId(String id) {
210    this.id = id;
211  }
212
213  /**
214   * @param name the name to set
215   */
216  public void setName(String name) {
217    this.name = name;
218  }
219
220  /**
221   * @param org the org to set
222   */
223  public void setOrg(OrganizationImpl org) {
224    this.org = org;
225  }
226
227  /**
228   * @param pk the pk to set
229   */
230  @SuppressWarnings("unused")
231  private void setPk(Long pk) {
232    this.pk = pk;
233  }
234
235  /**
236   * @param type the type to set
237   */
238  public void setType(MeasurementTypeImpl type) {
239    this.type = type;
240  }
241
242  /**
243   * @return the Depository equivalent to this.
244   */
245  public Depository toDepository() {
246    Depository ret = new Depository(id, name, type.toMeasurementType(), org.getId());
247    return ret;
248  }
249
250  /*
251   * (non-Javadoc)
252   * 
253   * @see java.lang.Object#toString()
254   */
255  @Override
256  public String toString() {
257    return "DepositoryImpl [pk=" + pk + ", id=" + id + ", name=" + name + ", type=" + type
258        + ", org=" + org + "]";
259  }
260
261}
