019package org.wattdepot.client.http.api.collector;
020
021import java.util.Date;
022import java.util.Timer;
023import java.util.TimerTask;
024
025import javax.xml.datatype.XMLGregorianCalendar;
026
027import org.apache.commons.cli.CommandLine;
028import org.apache.commons.cli.CommandLineParser;
029import org.apache.commons.cli.HelpFormatter;
030import org.apache.commons.cli.Options;
031import org.apache.commons.cli.ParseException;
032import org.apache.commons.cli.PosixParser;
033import org.wattdepot.client.http.api.WattDepotClient;
034import org.wattdepot.common.domainmodel.CollectorProcessDefinition;
035import org.wattdepot.common.domainmodel.Depository;
036import org.wattdepot.common.domainmodel.InterpolatedValue;
037import org.wattdepot.common.domainmodel.MeasurementList;
038import org.wattdepot.common.domainmodel.Sensor;
039import org.wattdepot.common.exception.BadCredentialException;
040import org.wattdepot.common.exception.BadSensorUriException;
041import org.wattdepot.common.exception.IdNotFoundException;
042import org.wattdepot.common.util.DateConvert;
043import org.wattdepot.common.util.tstamp.Tstamp;
044
045/**
046 * MeasurementSummaryClient polls the WattDepot server asking for the all the
047 * measurements in a depository.
048 * 
049 * @author Cam Moore
050 * 
051 */
052public class MeasurementSummaryClient extends TimerTask {
053  /** Flag for debugging messages. */
054  private boolean debug;
055  /** The client used to communicate with the WattDepot server. */
056  private WattDepotClient client;
057  /** The Depository for storing measurements. */
058  private Depository depository;
059  /** The Sensor making the measurements. */
060  private Sensor sensor;
061  /** The time of the earliest measurement. */
062  private Date earliest;
063  /** Flag for using a window. */
064  private boolean windowP;
065
066  /**
067   * @param serverUri The URI for the WattDepot server.
068   * @param username The name of a user defined in the WattDepot server.
069   * @param orgId the id of the organization the user is in.
070   * @param password The password for the user.
071   * @param collectorId The CollectorProcessDefinitionId used to initialize this
072   *        collector.
073   * @param debug flag for debugging messages.
074   * @throws BadCredentialException if the user or password don't match the
075   *         credentials in WattDepot.
076   * @throws IdNotFoundException if the processId is not defined.
077   * @throws BadSensorUriException if the Sensor's URI isn't valid.
078   */
079  public MeasurementSummaryClient(String serverUri, String username, String orgId, String password,
080      String collectorId, boolean debug) throws BadCredentialException, IdNotFoundException,
081      BadSensorUriException {
082    this.client = new WattDepotClient(serverUri, username, orgId, password);
083    this.debug = debug;
084    CollectorProcessDefinition definition = client.getCollectorProcessDefinition(collectorId);
085    this.depository = client.getDepository(definition.getDepositoryId());
086    this.sensor = client.getSensor(definition.getSensorId());
087    InterpolatedValue v = client.getEarliestValue(this.depository, sensor);
088    this.earliest = v.getDate();
089  }
090
091  /**
092   * @param args command line arguments.
093   */
094  public static void main(String[] args) {
095    Options options = new Options();
096    options.addOption("h", false, "Usage: MeasurementSummaryClient -s <server uri> -u <username>"
097        + " -p <password> -o <orgId> -c <collectorId> -m <millisecond sleep> [-d]");
098    options.addOption("s", "server", true, "WattDepot Server URI. (http://server.wattdepot.org)");
099    options.addOption("u", "username", true, "Username");
100    options.addOption("o", "organizationId", true, "User's Organization id.");
101    options.addOption("p", "password", true, "Password");
102    options.addOption("c", "collector", true, "Collector Process Definition Id");
103    options.addOption("m", "milliseconds", true, "Number of milliseconds to sleep between polls.");
104    options.addOption("d", "debug", false, "Displays sensor data as it is sent to the server.");
105    options.addOption("w", "windowing", false,
106        "Use a window instead of getting all the measurements.");
107
108    CommandLine cmd = null;
109    String serverUri = null;
110    String username = null;
111    String organizationId = null;
112    String password = null;
113    String collectorId = null;
114    Integer milliSeconds = null;
115    boolean debug = false;
116    boolean windowP = false;
117
118    CommandLineParser parser = new PosixParser();
119    HelpFormatter formatter = new HelpFormatter();
120    try {
121      cmd = parser.parse(options, args);
122    }
123    catch (ParseException e) {
124      System.err.println("Command line parsing failed. Reason: " + e.getMessage() + ". Exiting.");
125      System.exit(1);
126    }
127    if (cmd.hasOption("h")) {
128      formatter.printHelp("StressTestCollector", options);
129      System.exit(0);
130    }
131    if (cmd.hasOption("s")) {
132      serverUri = cmd.getOptionValue("s");
133    }
134    else {
135      serverUri = "http://server.wattdepot.org/";
136    }
137    if (cmd.hasOption("u")) {
138      username = cmd.getOptionValue("u");
139    }
140    else {
141      username = "user";
142    }
143    if (cmd.hasOption("p")) {
144      password = cmd.getOptionValue("p");
145    }
146    else {
147      password = "default";
148    }
149    if (cmd.hasOption("o")) {
150      organizationId = cmd.getOptionValue("o");
151    }
152    else {
153      organizationId = "organization";
154    }
155    if (cmd.hasOption("c")) {
156      collectorId = cmd.getOptionValue("c");
157    }
158    else {
159      collectorId = "stress_test";
160    }
161    if (cmd.hasOption("m")) {
162      milliSeconds = Integer.parseInt(cmd.getOptionValue("m"));
163    }
164    else {
165      milliSeconds = 1000;
166    }
167    debug = cmd.hasOption("d");
168    windowP = cmd.hasOption("w");
169    if (debug) {
170      System.out.println("Measurement Summary Client:");
171      System.out.println("    WattDepotServer: " + serverUri);
172      System.out.println("    Username: " + username);
173      System.out.println("    OrganizationId: " + organizationId);
174      System.out.println("    Password :" + password);
175      System.out.println("    CPD Id: " + collectorId);
176      System.out.println("    MilliSeconds: " + milliSeconds);
177      System.out.println("    Windowing: " + windowP);
178    }
179    Timer t = new Timer();
180    try {
181      MeasurementSummaryClient c = new MeasurementSummaryClient(serverUri, username,
182          organizationId, password, collectorId, debug);
183      c.windowP = windowP;
184      System.out.format("Started MeasurementSummaryClient at %s%n", Tstamp.makeTimestamp());
185      t.schedule(c, 0, milliSeconds);
186    }
187    catch (BadCredentialException e) {
188      e.printStackTrace();
189    }
190    catch (IdNotFoundException e) {
191      e.printStackTrace();
192    }
193    catch (BadSensorUriException e) {
194      e.printStackTrace();
195    }
196  }
197
198  /*
199   * (non-Javadoc)
200   * 
201   * @see java.util.TimerTask#run()
202   */
203  @Override
204  public void run() {
205    Date now = new Date();
206    MeasurementList meas = null;
207    if (windowP) {
208      XMLGregorianCalendar end = Tstamp.makeTimestamp(now.getTime());
209      XMLGregorianCalendar start = Tstamp.incrementSeconds(end, -30);
210      meas = client.getMeasurements(depository, sensor, DateConvert.convertXMLCal(start),
211          DateConvert.convertXMLCal(end));
212      if (debug) {
213        System.out
214            .println("At " + end + " there are " + meas.getMeasurements().size()
215                + " measurements in the window. Query took "
216                + Tstamp.diff(end, Tstamp.makeTimestamp()));
217      }
218    }
219    else {
220      meas = client.getMeasurements(depository, sensor, this.earliest, now);
221      if (debug) {
222        XMLGregorianCalendar start = Tstamp.makeTimestamp(now.getTime());
223        XMLGregorianCalendar end = Tstamp.makeTimestamp();
224        System.out.println("Query started at " + start + " there are "
225            + meas.getMeasurements().size() + " measurements. Query ended at " + end + ". Took "
226            + Tstamp.diff(start, end));
227      }
228    }
229  }
