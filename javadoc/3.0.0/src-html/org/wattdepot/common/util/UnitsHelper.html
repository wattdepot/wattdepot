019package org.wattdepot.common.util;
020
021import java.io.UnsupportedEncodingException;
022import java.text.Normalizer;
023import java.util.HashMap;
024import java.util.Map;
025import java.util.regex.Pattern;
026
027import javax.measure.unit.NonSI;
028import javax.measure.unit.SI;
029import javax.measure.unit.Unit;
030import javax.measure.unit.UnitFormat;
031
032/**
033 * UnitsHelper - Utility class that helps build measurement types.
034 * 
035 * @author Cam Moore
036 * 
037 */
038public class UnitsHelper {
039  /** Holds the defined units. */
040  public static final Map<String, Unit<?>> quantities = new HashMap<String, Unit<?>>();
041
042  static {
043    UnitFormat format = UnitFormat.getInstance();
044    format.alias(SI.MICRO(Unit.ONE), "ppm");
045    format.label(SI.MICRO(Unit.ONE), "ppm");    
046    quantities.put(buildName("Power", SI.WATT), SI.WATT);
047    quantities.put(buildName("Energy", SI.WATT.times(NonSI.HOUR)), SI.WATT.times(NonSI.HOUR));
048    quantities.put(buildName("Frequency", SI.HERTZ), SI.HERTZ);
049    quantities.put(buildName("Temperature", NonSI.FAHRENHEIT), NonSI.FAHRENHEIT);
050    quantities.put(buildName("Temperature", SI.CELSIUS), SI.CELSIUS);
051    quantities.put(buildName("Volume", NonSI.GALLON_LIQUID_US), NonSI.GALLON_LIQUID_US);
052    quantities.put(buildName("Volume", NonSI.LITER), NonSI.LITER);
053    quantities.put(buildName("Flow Rate", NonSI.GALLON_LIQUID_US.divide(SI.SECOND)),
054        NonSI.GALLON_LIQUID_US.divide(SI.SECOND));
055    quantities.put(buildName("Flow Rate", NonSI.LITER.divide(SI.SECOND)),
056        NonSI.LITER.divide(SI.SECOND));
057    quantities.put(buildName("Mass", SI.KILOGRAM), SI.KILOGRAM);
058    quantities.put(buildName("Mass", NonSI.POUND), NonSI.POUND);
059    quantities.put(buildName("Humidity", NonSI.PERCENT), NonSI.PERCENT);
060    quantities.put(buildName("Concentration", SI.MICRO(Unit.ONE)), SI.MICRO(Unit.ONE));
061    quantities.put(buildName("Cloud Coverage", NonSI.PERCENT), NonSI.PERCENT);
062
063  }
064
065  /**
066   * @param type
067   *          The type of the unit, Energy, Power, Mass, etc.
068   * @param unit
069   *          The Unit<?>.
070   * @return The name.
071   */
072  public static String buildName(String type, Unit<?> unit) {
073    String s = unit.toString();
074    String s1 = Normalizer.normalize(s, Normalizer.Form.NFKD);
075    String regex = Pattern.quote("[\\p{InCombiningDiacriticalMarks}\\p{IsLm}\\p{IsSk}]+");
076
077    try {
078      String s2 = new String(s1.replaceAll(regex, "").getBytes("ascii"), "ascii");
079      s2 = s2.replace("?", "");
080      return type + " (" + s2 + ")";
081      
082    }
083    catch (UnsupportedEncodingException e) {
084      e.printStackTrace();
085    }
086    return type + " (" + s1 + ")";
087  }
088
089
090}
