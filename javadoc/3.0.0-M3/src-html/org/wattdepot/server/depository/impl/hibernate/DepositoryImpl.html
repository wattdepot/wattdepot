058   * (non-Javadoc)

062  @Override
063  public boolean equals(Object obj) {
064    if (this == obj) {
065      return true;
066    }
067    if (!super.equals(obj)) {
068      return false;
069    }
070    if (getClass() != obj.getClass()) {
071      return false;
072    }
073    DepositoryImpl other = (DepositoryImpl) obj;
074    if (serverProperties == null) {
075      if (other.serverProperties != null) {
076        return false;
077      }
078    }
079    else if (!serverProperties.equals(other.serverProperties)) {
080      return false;
081    }
082    return true;
083  }

140    Measurement ret = null;
141    Session session = Manager.getFactory(serverProperties).openSession();
142    session.beginTransaction();
143    @SuppressWarnings("unchecked")
144    List<MeasurementImpl> measurements = (List<MeasurementImpl>) session
145        .createQuery("FROM MeasurementImpl").list();
146    for (MeasurementImpl meas : measurements) {
147      if (meas.getId().equals(measId)) {
148        ret = meas;
149      }
150    }
151    session.getTransaction().commit();
152    session.close();    
153    return ret;
154  }

180   *          The end of the interval.
181   * @return A list of the measurements in the interval.
182   */
183  @Override
184  public List<Measurement> getMeasurements(Sensor sensor, Date start, Date end) {
185    List<Measurement> ret = new ArrayList<Measurement>();
186    Session session = Manager.getFactory(serverProperties).openSession();
187    session.beginTransaction();
188    @SuppressWarnings("unchecked")
189    List<MeasurementImpl> measurements = (List<MeasurementImpl>) session
190        .createQuery(
191            "FROM MeasurementImpl WHERE date >= :start AND date <= :end AND depository = :depository")
192        .setParameter("start", start).setParameter("end", end)
193        .setParameter("depository", this).list();
194    for (MeasurementImpl meas : measurements) {
195      if (meas.getSensor().equals(sensor)) {
196        ret.add(meas);
197      }
198    }
199    session.getTransaction().commit();
200    session.close();
201    return ret;
202  }

234  /**
235   * @param sensor
236   *          The Sensor making the measurements.
237   * @param timestamp
238   *          The time of the value.
239   * @return The Value 'measured' at the given time, most likely an interpolated
240   *         value.
241   * @throws NoMeasurementException
242   *           If there aren't any measurements around the time.
243   */
244  public Double getValue(Sensor sensor, Date timestamp)
245      throws NoMeasurementException {
246    Double ret = null;
247    Session session = Manager.getFactory(serverProperties).openSession();
248    session.beginTransaction();
249
250    @SuppressWarnings("unchecked")
251    List<MeasurementImpl> result = (List<MeasurementImpl>) session
252        .createQuery(
253            "FROM MeasurementImpl WHERE date = :time AND measurementType = :measType"
254                + " AND depository = :name").setParameter("time", timestamp)
255        .setParameter("measType", getMeasurementType().getUnits())
256        .setParameter("name", this).list();
257    if (result.size() > 0) {
258      for (MeasurementImpl meas : result) {
259        if (meas.getSensor().equals(sensor)) {
260          ret = meas.getValue();
261        }
262      }
263    }
264    else {
265      // need to get the stradle
266      @SuppressWarnings("unchecked")
267      List<MeasurementImpl> before = (List<MeasurementImpl>) session
268          .createQuery(
269              "FROM MeasurementImpl WHERE date <= :time AND measurementType = :measType"
270                  + " AND depository = :name").setParameter("time", timestamp)
271          .setParameter("measType", getMeasurementType().getUnits())
272          .setParameter("name", this).list();
273      @SuppressWarnings("unchecked")
274      List<MeasurementImpl> after = (List<MeasurementImpl>) session
275          .createQuery(
276              "FROM MeasurementImpl WHERE date >= :time AND measurementType = :measType"
277                  + " AND depository = :name").setParameter("time", timestamp)
278          .setParameter("measType", getMeasurementType().getUnits())
279          .setParameter("name", this).list();
280      MeasurementImpl justBefore = null;
281      for (MeasurementImpl b : before) {
282        if (b.getSensor().equals(sensor)) {
283          if (justBefore == null) {
284            justBefore = b;
285          }
286          else if (b.getDate().compareTo(justBefore.getDate()) > 0) {
287            justBefore = b;
288          }
289        }
290        MeasurementImpl justAfter = null;
291        for (MeasurementImpl a : after) {
292          if (a.getSensor().equals(sensor)) {
293            if (justAfter == null) {
294              justAfter = a;
295            }
296            else if (a.getDate().compareTo(justBefore.getDate()) > 0) {
297              justAfter = a;
298            }
299          }
300          if (justBefore != null && justAfter != null) {
301            Double val1 = justBefore.getValue();
302            Double val2 = justAfter.getValue();
303            Double deltaV = val2 - val1;
304            Long t1 = justBefore.getDate().getTime();
305            Long t2 = justAfter.getDate().getTime();
306            Long deltaT = t2 - t1;
307            Long t3 = timestamp.getTime();
308            Long toDate = t3 - t1;
309            Double slope = deltaV / deltaT;
310            ret = val1 + (slope * toDate);
311          }
312          else if (justBefore == null && justAfter == null) {
313            session.getTransaction().commit();
314            session.close();
315            throw new NoMeasurementException(
316                "Cannot find measurements before or after " + timestamp);
317          }
318          else if (justBefore == null) {
319            session.getTransaction().commit();
320            session.close();
321            throw new NoMeasurementException("Cannot find measurement before "
322                + timestamp);
323          }
324          else if (justAfter == null) {
325            session.getTransaction().commit();
326            session.close();
327            throw new NoMeasurementException("Cannot find measurement after "
328                + timestamp);
329          }
330        }
331      }
332    }
333    session.getTransaction().commit();
334    session.close();
335    return ret;
336  }

338  /**
339   * @param sensor
340   *          The Sensor making the measurements.
341   * @param start
342   *          The start of the period.
343   * @param end
344   *          The end of the period.
345   * @return The value measured the difference between the end value and the
346   *         start value.
347   * @throws NoMeasurementException
348   *           if there are no measurements around the start or end time.
349   */
350  public Double getValue(Sensor sensor, Date start, Date end)
351      throws NoMeasurementException {
352    Double endVal = getValue(sensor, end);
353    Double startVal = getValue(sensor, start);
354    if (endVal != null && startVal != null) {
355      return endVal - startVal;
356    }
357    return null;
358  }

360  /**
361   * @param sensor
362   *          The Sensor making the measurements.
363   * @param start
364   *          The start of the interval.
365   * @param end
366   *          The end of the interval
367   * @param gapSeconds
368   *          The maximum number of seconds that measurements need to be within
369   *          the start and end.
370   * @return The value measured the difference between the end value and the
371   *         start value.
372   * @throws NoMeasurementException
373   *           if there are no measurements around the start or end time.
374   * @throws MeasurementGapException
375   *           if the measurements around start or end are too far apart.
376   */
377  public Double getValue(Sensor sensor, Date start, Date end, Long gapSeconds)
378      throws NoMeasurementException, MeasurementGapException {
379    Double endVal = getValue(sensor, end, gapSeconds);
380    Double startVal = getValue(sensor, start, gapSeconds);
381    if (endVal != null && startVal != null) {
382      return endVal - startVal;
383    }
384    return null;
385  }

402  public Double getValue(Sensor sensor, Date timestamp, Long gapSeconds)
403      throws NoMeasurementException, MeasurementGapException {
404    Double ret = null;

418          ret = meas.getValue();

435                  + " AND depository = :name").setParameter("time", timestamp)

447        }
448        MeasurementImpl justAfter = null;
449        for (MeasurementImpl a : after) {

457          }
458          if (justBefore != null && justAfter != null) {
459            Double val1 = justBefore.getValue();
460            Double val2 = justAfter.getValue();
461            Double deltaV = val2 - val1;
462            Long t1 = justBefore.getDate().getTime();
463            Long t2 = justAfter.getDate().getTime();
464            Long deltaT = t2 - t1;
465            if ((deltaT / 1000) > gapSeconds) {
466              session.getTransaction().commit();
467              session.close();
468              throw new MeasurementGapException("Gap of " + (deltaT / 1000)
469                  + "s is longer than " + gapSeconds);
470            }
471            Long t3 = timestamp.getTime();
472            Long toDate = t3 - t1;
473            Double slope = deltaV / deltaT;
474            ret = val1 + (slope * toDate);

485            throw new NoMeasurementException("Cannot find measurement before "
486                + timestamp);
487          }
488          else if (justAfter == null) {
489            session.getTransaction().commit();
490            session.close();
491            throw new NoMeasurementException("Cannot find measurement after "
492                + timestamp);
493          }
494        }
495      }
496    }
497    session.getTransaction().commit();
498    session.close();
499    return ret;

545  /**
546   * @param meas
547   *          The measurement to store.
548   * @throws MeasurementTypeException
549   *           if the type of the measurement doesn't match the Depository
550   *           measurement type.
551   */
552  @SuppressWarnings("unchecked")
553  @Override
554  public void putMeasurement(Measurement meas) throws MeasurementTypeException {
555    if (!meas.getMeasurementType().equals(getMeasurementType().getUnits())) {
556      throw new MeasurementTypeException("Measurement's type "
557          + meas.getMeasurementType() + " does not match "
558          + getMeasurementType());
559    }
560
561    Session session = Manager.getFactory(serverProperties).openSession();
562    session.beginTransaction();
563    // check the sensor
564    boolean haveSensor = false;
565    for (Sensor s : (List<Sensor>) session.createQuery("from Sensor").list()) {
566      if (s.getId().equals(meas.getSensor().getId())) {
567        haveSensor = true;
568      }
569    }
570    if (!haveSensor) {
571      // ensure the sensor has an owner
572      Sensor measSensor = meas.getSensor();
573      if (measSensor.getOwner() == null) {
574        measSensor.setOwner(getOwner());
575      }
576      saveSensor(session, measSensor);
577    }
578    MeasurementImpl mImpl = new MeasurementImpl(meas);
579    mImpl.setDepository(this);
580    session.saveOrUpdate(mImpl);
581    session.getTransaction().commit();
582    session.close();
583  }

585  /**
586   * Use this method after beginning a transaction.
587   * 
588   * @param session
589   *          The Session, a transaction must be in progress.
590   * @param sensor
591   *          The Sensor to save.
592   */
593  private void saveSensor(Session session, Sensor sensor) {
594    for (Property p : sensor.getProperties()) {
595      session.saveOrUpdate(p);
596    }
597    session.saveOrUpdate(sensor.getSensorLocation());
598    session.saveOrUpdate(sensor.getModel());
599    session.saveOrUpdate(sensor);
600  }
