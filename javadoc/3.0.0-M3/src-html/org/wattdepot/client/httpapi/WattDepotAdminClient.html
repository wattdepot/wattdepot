019package org.wattdepot.client.httpapi;
020
021import org.restlet.resource.ClientResource;
022import org.wattdepot.client.WattDepotAdminInterface;
023import org.wattdepot.common.domainmodel.UserGroup;
024import org.wattdepot.common.domainmodel.UserInfo;
025import org.wattdepot.common.domainmodel.UserPassword;
026import org.wattdepot.common.exception.BadCredentialException;
027import org.wattdepot.common.exception.IdNotFoundException;
028import org.wattdepot.common.httpapi.API;
029import org.wattdepot.common.httpapi.UserGroupResource;
030import org.wattdepot.common.httpapi.UserInfoResource;
031import org.wattdepot.common.httpapi.UserPasswordResource;
032
033/**
034 * WattDepotAdminClient - Admin level client.
035 * 
036 * @author Cam Moore
037 * 
038 */
039public class WattDepotAdminClient extends WattDepotClient implements WattDepotAdminInterface {
040
041  /**
042   * Creates a new WattDepotAdminClient.
043   * 
044   * @param serverUri
045   *          The URI of the WattDepot server (e.g.
046   *          "http://server.wattdepot.org/")
047   * @param username
048   *          The name of the user. The user must be defined in the WattDepot
049   *          server.
050   * @param password
051   *          The password for the user.
052   * @throws BadCredentialException
053   *           If the user or password don't match the credentials on the
054   *           WattDepot server.
055   */
056  public WattDepotAdminClient(String serverUri, String username, String password)
057      throws BadCredentialException {
058    super(serverUri, username, password);
059    if (!getGroupId().equals("admin")) {
060      throw new BadCredentialException("Wrong group.");
061    }
062  }
063
064  /*
065   * (non-Javadoc)
066   * 
067   * @see
068   * org.wattdepot.client.WattDepotAdminInterface#deleteUser(java.lang.String)
069   */
070  @Override
071  public void deleteUser(String id) throws IdNotFoundException {
072    ClientResource client = makeClient(getGroupId() + "/" + API.USER_URI + id);
073    UserInfoResource resource = client.wrap(UserInfoResource.class);
074    resource.remove();
075    client.release();
076  }
077
078  /*
079   * (non-Javadoc)
080   * 
081   * @see
082   * org.wattdepot.client.WattDepotAdminInterface#deleteUserGroup(java.lang
083   * .String)
084   */
085  @Override
086  public void deleteUserGroup(String id) throws IdNotFoundException {
087    ClientResource client = makeClient(getGroupId() + "/" + API.USER_GROUP_URI + id);
088    UserGroupResource resource = client.wrap(UserGroupResource.class);
089    resource.remove();
090    client.release();
091  }
092
093  /* (non-Javadoc)
094   * @see org.wattdepot.client.WattDepotAdminInterface#deleteUserPassword(java.lang.String)
095   */
096  @Override
097  public void deleteUserPassword(String id) throws IdNotFoundException {
098    ClientResource client = makeClient(getGroupId() + "/" + API.USER_PASSWORD_URI + id);
099    UserPasswordResource resource = client.wrap(UserPasswordResource.class);
100    resource.remove();
101    client.release();
102  }
103
104  /*
105   * (non-Javadoc)
106   * 
107   * @see
108   * org.wattdepot.client.WattDepotAdminInterface#putUser(org.wattdepot.datamodel
109   * .UserInfo)
110   */
111  @Override
112  public void putUser(UserInfo user) {
113    ClientResource client = makeClient(getGroupId() + "/" + API.USER_URI + user.getId());
114    UserInfoResource resource = client.wrap(UserInfoResource.class);
115    resource.store(user);
116    client.release();
117  }
118
119  /*
120   * (non-Javadoc)
121   * 
122   * @see
123   * org.wattdepot.client.WattDepotAdminInterface#putUserGroup(org.wattdepot
124   * .datamodel.UserGroup)
125   */
126  @Override
127  public void putUserGroup(UserGroup group) {
128    ClientResource client = makeClient(getGroupId() + "/" + API.USER_GROUP_URI + group.getId());
129    UserGroupResource resource = client.wrap(UserGroupResource.class);
130    resource.store(group);
131    client.release();
132  }
133
134  /*
135   * (non-Javadoc)
136   * 
137   * @see
138   * org.wattdepot.client.WattDepotAdminInterface#putUserPassword(org.wattdepot
139   * .datamodel.UserPassword)
140   */
141  @Override
142  public void putUserPassword(UserPassword password) {
143    ClientResource client = makeClient(getGroupId() + "/" + API.USER_PASSWORD_URI + password.getId());
144    UserPasswordResource resource = client.wrap(UserPasswordResource.class);
145    resource.store(password);
146    client.release();
147  }
148
149}
