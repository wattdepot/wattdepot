019package org.wattdepot.common.domainmodel;
020
021import java.util.HashSet;
022import java.util.Set;
023
024import org.wattdepot.common.util.Slug;
025
026/**
027 * Organization - A group of users used to create a namespace for WattDepot
028 * entities. Users, sensors, sensor groups, collectors, depositories, and
029 * measurements all belong to a single "organization".
030 * 
031 * @author Cam Moore
032 * 
033 */
034public class Organization {
035  /** The name of the admin group. */
036  public static final String ADMIN_GROUP_NAME = "admin";
037  /** The admin user group. */
038  public static final Organization ADMIN_GROUP = new Organization(ADMIN_GROUP_NAME);
039
040  /** The public user group. All groups can see publicly owned objects. */
041  public static final Organization PUBLIC_GROUP = new Organization(Labels.PUBLIC);
042
043  /** Unique id that is also a slug usable in URLs. */
044  protected String id;
045  /** The name of the group. */
046  protected String name;
047  /** The ids of the users in this group. */
048  protected Set<String> users;
049
050  static {
051    ADMIN_GROUP.add(UserInfo.ROOT.getUid());
052  }
053
054  /**
055   * The default constructor.
056   */
057  public Organization() {
058    users = new HashSet<String>();
059  }
060
061  /**
062   * @param name
063   *          The name of the Organization.
064   */
065  public Organization(String name) {
066    this(Slug.slugify(name), name, new HashSet<String>());
067  }
068
069  /**
070   * @param name
071   *          The name of the organization.
072   * @param users
073   *          The Users in the organization.
074   */
075  public Organization(String name, Set<String> users) {
076    this(Slug.slugify(name), name, users);
077  }
078
079  /**
080   * @param slug
081   *          The unique slug.
082   * @param name
083   *          The name of the organization.
084   * @param users
085   *          The Users in the organization.
086   */
087  public Organization(String slug, String name, Set<String> users) {
088    this.id = slug;
089    this.name = name;
090    this.users = users;
091  }
092
093  /**
094   * @param userId
095   *          The id of the User to add.
096   * @return true if successful.
097   * @see java.util.List#add(java.lang.Object)
098   */
099  public boolean add(String userId) {
100    return users.add(userId);
101  }
102
103  /**
104   * @param o
105   *          The user to test.
106   * @return true if the user is in the group.
107   * @see java.util.List#contains(java.lang.Object)
108   */
109  public boolean contains(Object o) {
110    return users.contains(o);
111  }
112
113  /*
114   * (non-Javadoc)
115   * 
116   * @see java.lang.Object#equals(java.lang.Object)
117   */
118  @Override
119  public boolean equals(Object obj) {
120    if (this == obj) {
121      return true;
122    }
123    if (obj == null) {
124      return false;
125    }
126    if (!getClass().isAssignableFrom(obj.getClass())
127        && !obj.getClass().isAssignableFrom(getClass()) && getClass() != obj.getClass()) {
128      return false;
129    }
130    Organization other = (Organization) obj;
131    if (id == null) {
132      if (other.id != null) {
133        return false;
134      }
135    }
136    else if (!id.equals(other.id)) {
137      return false;
138    }
139    if (name == null) {
140      if (other.name != null) {
141        return false;
142      }
143    }
144    else if (!name.equals(other.name)) {
145      return false;
146    }
147    if (users == null) {
148      if (other.users != null) {
149        return false;
150      }
151    }
152    else if (!users.equals(other.users)) {
153      return false;
154    }
155    return true;
156  }
157
158  /**
159   * @return The unique id for the Organization.
160   */
161  public String getId() {
162    return id;
163  }
164
165  /**
166   * @return the name
167   */
168  public String getName() {
169    return name;
170  }
171
172  /**
173   * @return the users.
174   */
175  public Set<String> getUsers() {
176    return users;
177  }
178
179  /*
180   * (non-Javadoc)
181   * 
182   * @see java.lang.Object#hashCode()
183   */
184  @Override
185  public int hashCode() {
186    final int prime = 31;
187    int result = 1;
188    result = prime * result + ((id == null) ? 0 : id.hashCode());
189    result = prime * result + ((name == null) ? 0 : name.hashCode());
190    result = prime * result + ((users == null) ? 0 : users.hashCode());
191    return result;
192  }
193
194  /**
195   * @param o
196   *          The User to remove.
197   * @return true if successful.
198   * @see java.util.List#remove(java.lang.Object)
199   */
200  public boolean remove(Object o) {
201    return users.remove(o);
202  }
203
204  /**
205   * @param id
206   *          the id to set
207   */
208  public void setId(String id) {
209    this.id = id;
210  }
211
212  /**
213   * @param name
214   *          the name to set
215   */
216  public void setName(String name) {
217    this.name = name;
218    if (this.id == null) {
219      this.id = Slug.slugify(name);
220    }
221  }
222
223  /**
224   * @param users
225   *          the users to set
226   */
227  public void setUsers(Set<String> users) {
228    this.users = users;
229  }
230
231  /*
232   * (non-Javadoc)
233   * 
234   * @see java.lang.Object#toString()
235   */
236  @Override
237  public String toString() {
238    return "Organization [id=" + id + ", name=" + name + ", users=" + users + "]";
239  }
240
241}
