019package org.wattdepot.common.domainmodel;
020
021import javax.measure.unit.Unit;
022
023import org.wattdepot.common.exception.BadSlugException;
024import org.wattdepot.common.util.Slug;
025
026/**
027 * MeasurementType - Defines the type of a Measurement. This class includes a
028 * human readable name, slug, and JScience Unit<?>.
029 * 
030 * @author Cam Moore
031 * 
032 */
033public class MeasurementType {
034  /** The name of the MeasurementType. */
035  private String name;
036  /** The unique id for the MeasurementType. */
037  private String id;
038  /** The Units for the measurement type. */
039  private Unit<?> unit;
040  /** String property stored in persistence. */
041  private String units;
042
043  /**
044   * Default constructor.
045   */
046  public MeasurementType() {
047
048  }
049
050  /**
051   * Creates a new MeasurementType.
052   * 
053   * @param name the name of the type.
054   * @param unit the units of measurement.
055   */
056  public MeasurementType(String name, Unit<?> unit) {
057    this(Slug.slugify(name), name, unit);
058  }
059
060  /**
061   * Creates a new MeasurementType.
062   * 
063   * @param id The unique id.
064   * @param name the name of the type.
065   * @param unit the units of measurement.
066   */
067  public MeasurementType(String id, String name, Unit<?> unit) {
068    this.id = id;
069    this.name = name;
070    this.unit = unit;
071    this.units = this.unit.toString();
072  }
073
074  /*
075   * (non-Javadoc)
076   * 
077   * @see java.lang.Object#equals(java.lang.Object)
078   */
079  @Override
080  public boolean equals(Object obj) {
081    if (this == obj) {
082      return true;
083    }
084    if (obj == null) {
085      return false;
086    }
087    if (getClass() != obj.getClass()) {
088      return false;
089    }
090    MeasurementType other = (MeasurementType) obj;
091    if (name == null) {
092      if (other.name != null) {
093        return false;
094      }
095    }
096    else if (!name.equals(other.name)) {
097      return false;
098    }
099    if (id == null) {
100      if (other.id != null) {
101        return false;
102      }
103    }
104    else if (!id.equals(other.id)) {
105      return false;
106    }
107    if (units == null) {
108      if (other.units != null) {
109        return false;
110      }
111    }
112    else if (!units.equals(other.units)) {
113      return false;
114    }
115    return true;
116  }
117
118  /**
119   * @return the slug
120   */
121  public String getId() {
122    return id;
123  }
124
125  /**
126   * @return the name
127   */
128  public String getName() {
129    return name;
130  }
131
132  /**
133   * @return the units
134   */
135  public String getUnits() {
136    return units;
137  }
138
139  /*
140   * (non-Javadoc)
141   * 
142   * @see java.lang.Object#hashCode()
143   */
144  @Override
145  public int hashCode() {
146    final int prime = 31;
147    int result = 1;
148    result = prime * result + ((name == null) ? 0 : name.hashCode());
149    result = prime * result + ((id == null) ? 0 : id.hashCode());
150    result = prime * result + ((units == null) ? 0 : units.hashCode());
151    return result;
152  }
153
154  /**
155   * @param id the slug to set
156   * @exception BadSlugException if the id is not valid.
157   */
158  public void setId(String id) throws BadSlugException {
159    if (Slug.validateSlug(id)) {
160      this.id = id;
161    }
162    else {
163      throw new BadSlugException(id + " is not a valid slug.");
164    }
165  }
166
167  /**
168   * @param name the name to set
169   */
170  public void setName(String name) {
171    this.name = name;
172    if (this.id == null) {
173      this.id = Slug.slugify(name);
174    }
175  }
176
177  /**
178   * @param units the units to set
179   */
180  public void setUnits(String units) {
181    this.units = units;
182    this.unit = Unit.valueOf(units);
183  }
184
185  /*
186   * (non-Javadoc)
187   * 
188   * @see java.lang.Object#toString()
189   */
190  @Override
191  public String toString() {
192    return "MeasurementType [id=" + id + ", name=" + name + ", units=" + units + "]";
193  }
194
195  /**
196   * @return The Unit<?> associated with this MeasurementType.
197   */
198  public Unit<?> unit() {
199    return unit;
200  }
201}
