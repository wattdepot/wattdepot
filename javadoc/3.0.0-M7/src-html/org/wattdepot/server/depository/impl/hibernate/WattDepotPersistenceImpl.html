254    }

256      CollectorProcessDefinition cpd = getCollectorProcessDefinition(id, orgId);
257      if (cpd != null) {
258        throw new UniqueIdException(id + " is already a CollectorProcessDefinition id.");
259      }
260    }
261    catch (IdNotFoundException e) { // NOPMD
262      // this is expected.
263    }
264    Session session = Manager.getFactory(getServerProperties()).openSession();
265    sessionOpen++;
266    session.beginTransaction();
267    OrganizationImpl org = retrieveOrganization(session, orgId);
268    SensorImpl sensor = retrieveSensor(session, sensorId, orgId);
269    DepositoryImpl depot = retrieveDepository(session, depositoryId, orgId);
270    Set<PropertyImpl> props = new HashSet<PropertyImpl>();
271    for (Property p : properties) {
272      props.add(new PropertyImpl(p));
273    }
274    CollectorProcessDefinitionImpl impl = new CollectorProcessDefinitionImpl(id, name, sensor,
275        pollingInterval, depot, props, org);
276    storeCollectorProcessDefinition(session, impl);
277    session.getTransaction().commit();
278    session.close();
279    sessionClose++;
280    return impl.toCPD();

467    sessionOpen++;
468    session.beginTransaction();

623  @Override
624  public void deleteOrganization(String id) throws IdNotFoundException {
625    getOrganization(id);
626    // Remove Organization owned CollectorProcessDefinitions
627    Session session = Manager.getFactory(getServerProperties()).openSession();
628    sessionOpen++;
629    session.beginTransaction();
630    for (CollectorProcessDefinitionImpl sp : retrieveCollectorProcessDefinitions(session, id)) {
631      session.delete(sp);
632    }
633    session.getTransaction().commit();
634    session.close();
635    sessionClose++;
636    // Remove Organization owned SensorGroups
637    session = Manager.getFactory(getServerProperties()).openSession();
638    sessionOpen++;
639    session.beginTransaction();
640    for (SensorGroupImpl sg : retrieveSensorGroups(session, id)) {
641      session.delete(sg);
642    }
643    session.getTransaction().commit();
644    session.close();
645    sessionClose++;
646    // Remove Organization owned Measurements
647    session = Manager.getFactory(getServerProperties()).openSession();
648    sessionOpen++;
649    session.beginTransaction();
650    for (DepositoryImpl d : retrieveDepositories(session, id)) {
651      for (String sensorId : listSensors(d.getId(), id)) {
652        for (Measurement m : getMeasurements(d.getId(), id, sensorId)) {
653          MeasurementImpl mi = retrieveMeasurement(session, d.getId(), id, m.getId());
654          if (mi != null) {
655            session.delete(mi);
656          }
657        }
658      }
659    }
660    session.getTransaction().commit();
661    session.close();
662    sessionClose++;
663    // Remove Organization owned Depositories and Sensors
664    session = Manager.getFactory(getServerProperties()).openSession();
665    sessionOpen++;
666    session.beginTransaction();
667    List<DepositoryImpl> depositories = retrieveDepositories(session, id);
668    session.getTransaction().commit();
669    session.close();
670    sessionClose++;
671    session = Manager.getFactory(getServerProperties()).openSession();
672    sessionOpen++;
673    session.beginTransaction();
674    for (DepositoryImpl d : depositories) {
675      session.delete(d);
676    }
677    for (SensorImpl s : retrieveSensors(session, id)) {
678      session.delete(s);
679    }
680    session.getTransaction().commit();
681    session.close();
682    sessionClose++;
683    // Remove Organization owned SensorModels
684    session = Manager.getFactory(getServerProperties()).openSession();
685    sessionOpen++;
686    session.beginTransaction();
687    for (SensorModelImpl sm : retrieveSensorModels(session)) {
688      if (!SensorModelHelper.models.containsKey(sm.getName())) {
