019package org.wattdepot.common.util;
020
021import org.wattdepot.common.domainmodel.Labels;
022import org.wattdepot.common.util.tstamp.Tstamp;
023
024import java.text.ParseException;
025import java.text.SimpleDateFormat;
026import java.util.Date;
027import java.util.GregorianCalendar;
028
029import javax.xml.datatype.DatatypeConfigurationException;
030import javax.xml.datatype.DatatypeFactory;
031import javax.xml.datatype.XMLGregorianCalendar;
032
033/**
034 * DateUtilities - Utility functions dealing with java.util.Date and
035 * java.xml.datatype.XMLGregorianCalendar instances.
036 *
037 * @author Cam Moore
038 *
039 */
040public class DateConvert {
041
042  /**
043   * Parses the given String representation of an XMLGregorianCalendar and
044   * returns the instance.
045   *
046   * @param s
047   *          The String representation of an XMLGregorianCalendar with the
048   *          format yyyy-MM-dd'T'HH:mm:ss.SSS
049   * @return The XMLGregorianCalendar instance.
050   * @throws ParseException
051   *           if the string has the wrong format.
052   * @throws DatatypeConfigurationException
053   *           if there is a problems instantiating the DatatypeFactory.
054   */
055  public static XMLGregorianCalendar parseCalString(String s) throws ParseException,
056      DatatypeConfigurationException {
057    XMLGregorianCalendar result = null;
058    SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSX");
059    result = convertDate(simpleDateFormat.parse(s));
060    return result;
061  }
062
063  /**
064   * @param s
065   *          The String representation in the format yyyy-MM-dd'T'HH:mm:ss.SSS.
066   * @return The Date instance for the time.
067   * @throws ParseException
068   *           if the string has the wrong format.
069   * @throws DatatypeConfigurationException
070   *           if there is a problem initializing the DatatypeFactory.
071   */
072  public static Date parseCalStringToDate(String s) throws ParseException,
073      DatatypeConfigurationException {
074    return convertXMLCal(parseCalString(s));
075  }
076
077  /**
078   * Converts a java.util.Date to a javax.xml.datatype.XMLGregorianCalendar.
079   *
080   * @param date
081   *          The Date to convert.
082   * @return The XMLGregorianCalendar representation of the given Date.
083   * @throws DatatypeConfigurationException
084   *           if there is a problem instantiating the DatatypeFactory.
085   */
086  public static XMLGregorianCalendar convertDate(Date date) throws DatatypeConfigurationException {
087    XMLGregorianCalendar result = null;
088    GregorianCalendar gregorianCalendar = (GregorianCalendar) GregorianCalendar.getInstance();
089    gregorianCalendar.setTime(date);
090    result = DatatypeFactory.newInstance().newXMLGregorianCalendar(gregorianCalendar);
091    return result;
092  }
093
094  /**
095   * Converts the given XMLGregorianCalendar to a java.util.Date.
096   *
097   * @param xgc
098   *          The XMLGregorianCalendar.
099   * @return The Date representation of the XMLGregorianCalendar.
100   */
101  public static Date convertXMLCal(XMLGregorianCalendar xgc) {
102    return xgc.toGregorianCalendar().getTime();
103  }
104
105  /**
106   * Returns the beginning of the hour or day given the time and hour or daily choice.
107   * @param time              the current time.
108   * @param hourlyDailyChoice how wide the window should be.
109   * @return the beginning of the window based upon the current time and window width.
110   */
111  public static XMLGregorianCalendar getBeginning(XMLGregorianCalendar time, String hourlyDailyChoice) {
112    XMLGregorianCalendar begin = null;
113    if (hourlyDailyChoice.equals(Labels.HOURLY)) {
114      begin = Tstamp.makeTimestamp(time.toGregorianCalendar().getTimeInMillis());
115      begin.setMinute(0);
116      begin.setSecond(0);
117      begin.setMillisecond(0);
118    }
119    else if (hourlyDailyChoice.equals(Labels.DAILY)) {
120      begin = Tstamp.makeTimestamp(time.toGregorianCalendar().getTimeInMillis());
121      begin.setHour(0);
122      begin.setMinute(0);
123      begin.setSecond(0);
124      begin.setMillisecond(0);
125    }
126    return begin;
127  }
128
129  /**
130   * Returns the end of the hour or day give the time and the hour or daily choice.
131   * @param time              the current time.
132   * @param hourlyDailyChoice the width of the window.
133   * @return the end of the window based upon the current time and the window width.
134   */
135  public static XMLGregorianCalendar getEnding(XMLGregorianCalendar time, String hourlyDailyChoice) {
136    XMLGregorianCalendar end = null;
137    if (hourlyDailyChoice.equals(Labels.HOURLY)) {
138      end = Tstamp.incrementHours(time, 1);
139      end.setMinute(0);
140      end.setSecond(0);
141      end.setMillisecond(0);
142    }
143    else if (hourlyDailyChoice.equals(Labels.DAILY)) {
144      end = Tstamp.incrementDays(time, 1);
145      end.setHour(0);
146      end.setMinute(0);
147      end.setSecond(0);
148      end.setMillisecond(0);
149    }
150    return end;
151  }
