001package org.wattdepot.common.domainmodel;
002
003import java.util.Date;
004
005/**
006 * Two value InterpolatedValue.
007 * @author Cam Moore.
008 */
009public class XYInterpolatedValue extends InterpolatedValue {
010  /** The id of the sensor making the measurement. */
011  private String ySensorId;
012  /** The value of the measurement. */
013  private Double yValue;
014  /** The type of the measurement. */
015  private MeasurementType yMeasurementType;
016
017  /**
018   * Default constructor.
019   */
020  public XYInterpolatedValue() {
021    super();
022  }
023
024  /**
025   * Creates a new XYInterpolatedValue.
026   * @param xSensorId The x SensorId.
027   * @param xValue The x value.
028   * @param xMeasurementType The x measurement type.
029   * @param date The time of both values.
030   * @param ySensorId The y SensorId.
031   * @param yValue The y value.
032   * @param yMeasurementType The y measurement type.
033   */
034  public XYInterpolatedValue(String xSensorId, Double xValue, MeasurementType xMeasurementType, Date date,
035                             String ySensorId, Double yValue, MeasurementType yMeasurementType) {
036    super(xSensorId, xValue, xMeasurementType, date);
037    this.ySensorId = ySensorId;
038    this.yValue = yValue;
039    this.yMeasurementType = yMeasurementType;
040  }
041
042  /**
043   * Creates a new XYInterpolatedValue.
044   * @param xSensorId The x SensorId.
045   * @param xValue The x value.
046   * @param xMeasurementType The x measurement type.
047   * @param startDate The start time of both values.
048   * @param endDate The end time of both values.
049   * @param ySensorId The y SensorId.
050   * @param yValue The y value.
051   * @param yMeasurementType The y measurement type.
052   */
053  public XYInterpolatedValue(String xSensorId, Double xValue, MeasurementType xMeasurementType, Date startDate,
054                             Date endDate, String ySensorId, Double yValue, MeasurementType yMeasurementType) {
055    super(xSensorId, xValue, xMeasurementType, startDate, endDate);
056    this.ySensorId = ySensorId;
057    this.yValue = yValue;
058    this.yMeasurementType = yMeasurementType;
059  }
060
061  /**
062   * Create a new XYInterpolatedValue from the two Interpolated values.
063   * @param x The X value.
064   * @param y The Y value.
065   */
066  public XYInterpolatedValue(InterpolatedValue x, InterpolatedValue y) {
067    super(x.getSensorId(), x.getValue(), x.getMeasurementType(), x.getStart(), x.getEnd());
068    if (!x.getStart().equals(y.getStart()) ||
069      !x.getEnd().equals(y.getEnd())) {
070      throw new IllegalArgumentException("X and Y don't have the same start and end times.");
071    }
072    this.ySensorId = y.getSensorId();
073    this.yValue = y.getValue();
074    this.yMeasurementType = y.getMeasurementType();
075
076  }
077
078  @Override
079  public boolean equals(Object o) {
080    if (this == o) {
081      return true;
082    }
083    if (o == null || getClass() != o.getClass()) {
084      return false;
085    }
086    if (!super.equals(o)) {
087      return false;
088    }
089
090    XYInterpolatedValue that = (XYInterpolatedValue) o;
091
092    if (ySensorId != null ? !ySensorId.equals(that.ySensorId) : that.ySensorId != null) {
093      return false;
094    }
095    if (yValue != null ? !yValue.equals(that.yValue) : that.yValue != null) {
096      return false;
097    }
098    return !(yMeasurementType != null ? !yMeasurementType.equals(that.yMeasurementType) : that.yMeasurementType != null);
099
100  }
101
102  @Override
103  public int hashCode() {
104    int result = super.hashCode();
105    result = 31 * result + (ySensorId != null ? ySensorId.hashCode() : 0);
106    result = 31 * result + (yValue != null ? yValue.hashCode() : 0);
107    result = 31 * result + (yMeasurementType != null ? yMeasurementType.hashCode() : 0);
108    return result;
109  }
110
111  /**
112   * @return The Y SensorId.
113   */
114  public String getYSensorId() {
115    return ySensorId;
116  }
117
118  /**
119   * Sets the Y SensorID.
120   * @param ySensorId The new Y SensorId.
121   */
122  public void setYSensorId(String ySensorId) {
123    this.ySensorId = ySensorId;
124  }
125
126  /**
127   * @return The Y value.
128   */
129  public Double getYValue() {
130    return yValue;
131  }
132
133  /**
134   * Sets the Y value.
135   * @param yValue The new Y value.
136   */
137  public void setYValue(Double yValue) {
138    this.yValue = yValue;
139  }
140
141  /**
142   * @return The Y MeasurementType.
143   */
144  public MeasurementType getYMeasurementType() {
145    return yMeasurementType;
146  }
147
148  /**
149   * Sets the Y MeasurementType.
150   * @param yMeasurementType The new Y MeasurementType.
151   */
152  public void setYMeasurementType(MeasurementType yMeasurementType) {
153    this.yMeasurementType = yMeasurementType;
154  }
155
156  /**
157   * @return The X SensorId.
158   */
159  public String getXSensorId() {
160    return getSensorId();
161  }
162
163  /**
164   * Sets the X SensorId.
165   * @param xSensorId The new X SensorId.
166   */
167  public void setXSensorId(String xSensorId) {
168    this.setSensorId(xSensorId);
169  }
170
171  /**
172   * @return The X value.
173   */
174  public Double getXValue() {
175    return getValue();
176  }
177
178  /**
179   * Sets the X value.
180   * @param xValue The new X value.
181   */
182  public void setXValue(Double xValue) {
183    this.setValue(xValue);
184  }
185
186  /**
187   * @return The X MeasurementType.
188   */
189  public MeasurementType getXMeasurementType() {
190    return getMeasurementType();
191  }
192
193  /**
194   * Sets the X MeasurementType.
195   * @param xMeasurementType The new X MeasurementType.
196   */
197  public void setXMeasurementType(MeasurementType xMeasurementType) {
198    this.setMeasurementType(xMeasurementType);
199  }
200}
