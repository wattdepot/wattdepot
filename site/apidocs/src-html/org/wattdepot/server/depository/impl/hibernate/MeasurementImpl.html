019package org.wattdepot.server.depository.impl.hibernate;
020
021import java.util.Date;
022
023import javax.measure.unit.Unit;
024import javax.persistence.Column;
025import javax.persistence.Entity;
026import javax.persistence.GeneratedValue;
027import javax.persistence.Id;
028import javax.persistence.Index;
029import javax.persistence.ManyToOne;
030import javax.persistence.Table;
031import javax.persistence.Temporal;
032import javax.persistence.TemporalType;
033
034//import org.hibernate.annotations.Index;
035import org.wattdepot.common.domainmodel.Measurement;
036
037/**
038 * MeasurementImpl - Hibernate persistant version of Measurement. It is 'stored'
039 * in a Depository.
040 * 
041 * @author Cam Moore
042 * 
043 */
044@Entity
045@Table(name = "MEASUREMENTS", indexes = {@Index(name = "IDX_MEASUREMENT", columnList = "DEPOSITORY_PK, SENSOR_PK, TIMESTAMP") })
046public class MeasurementImpl {
047  /** Database primary key. */
048  @Id
049  @GeneratedValue
050  private Long pk;
051  /** The unique id. Can be used in URIs. */
052  private String id;
053  /** The sensor that made the measurement. */
054  @ManyToOne
055//  @Index(name = "IDX_MEASUREMENT")
056  private SensorImpl sensor;
057  /** The time of the measurement. */
058//  @Index(name = "IDX_MEASUREMENT")
059  private Date timestamp;
060  /** The value of the measurement. */
061  private Double value;
062  /** The units of the measurement. */
063  private String units;
064  /** The Depository storing the measurement. */
065  @ManyToOne
066//  @Index(name = "IDX_MEASUREMENT")
067  private DepositoryImpl depository;
068
069  /**
070   * Default constructor.
071   */
072  public MeasurementImpl() {
073    super();
074  }
075
076  /*
077   * (non-Javadoc)
078   * 
079   * @see java.lang.Object#equals(java.lang.Object)
080   */
081  @Override
082  public boolean equals(Object obj) {
083    if (this == obj) {
084      return true;
085    }
086    if (obj == null) {
087      return false;
088    }
089    if (getClass() != obj.getClass()) {
090      return false;
091    }
092    MeasurementImpl other = (MeasurementImpl) obj;
093    if (id == null) {
094      if (other.id != null) {
095        return false;
096      }
097    }
098    else if (!id.equals(other.id)) {
099      return false;
100    }
101    if (pk == null) {
102      if (other.pk != null) {
103        return false;
104      }
105    }
106    else if (!pk.equals(other.pk)) {
107      return false;
108    }
109    if (sensor == null) {
110      if (other.sensor != null) {
111        return false;
112      }
113    }
114    else if (!sensor.equals(other.sensor)) {
115      return false;
116    }
117    if (timestamp == null) {
118      if (other.timestamp != null) {
119        return false;
120      }
121    }
122    else if (!timestamp.equals(other.timestamp)) {
123      return false;
124    }
125    if (units == null) {
126      if (other.units != null) {
127        return false;
128      }
129    }
130    else if (!units.equals(other.units)) {
131      return false;
132    }
133    if (value == null) {
134      if (other.value != null) {
135        return false;
136      }
137    }
138    else if (!value.equals(other.value)) {
139      return false;
140    }
141    return true;
142  }
143
144  /**
145   * @return the depository
146   */
147  public DepositoryImpl getDepository() {
148    return depository;
149  }
150
151  /**
152   * @return the id
153   */
154  public String getId() {
155    return id;
156  }
157
158  /**
159   * @return the pk
160   */
161  public Long getPk() {
162    return pk;
163  }
164
165  /**
166   * @return the sensor
167   */
168  public SensorImpl getSensor() {
169    return sensor;
170  }
171
172  /**
173   * @return the timestamp
174   */
175  @Temporal(TemporalType.TIMESTAMP)
176  @Column(name = "MEASUREMENT_DATE")
177  public Date getTimestamp() {
178    return new Date(timestamp.getTime());
179  }
180
181  /**
182   * @return the units
183   */
184  public String getUnits() {
185    return units;
186  }
187
188  /**
189   * @return the value
190   */
191  public Double getValue() {
192    return value;
193  }
194
195  /*
196   * (non-Javadoc)
197   * 
198   * @see java.lang.Object#hashCode()
199   */
200  @Override
201  public int hashCode() {
202    final int prime = 31;
203    int result = 1;
204    result = prime * result + ((id == null) ? 0 : id.hashCode());
205    result = prime * result + ((pk == null) ? 0 : pk.hashCode());
206    result = prime * result + ((sensor == null) ? 0 : sensor.hashCode());
207    result = prime * result + ((timestamp == null) ? 0 : timestamp.hashCode());
208    result = prime * result + ((units == null) ? 0 : units.hashCode());
209    result = prime * result + ((value == null) ? 0 : value.hashCode());
210    return result;
211  }
212
213  /**
214   * @param depository the depository to set
215   */
216  public void setDepository(DepositoryImpl depository) {
217    this.depository = depository;
218  }
219
220  /**
221   * @param id the id to set
222   */
223  public void setId(String id) {
224    this.id = id;
225  }
226
227  /**
228   * @param pk the pk to set
229   */
230  @SuppressWarnings("unused")
231  private void setPk(Long pk) {
232    this.pk = pk;
233  }
234
235  /**
236   * @param sensor the sensor to set
237   */
238  public void setSensor(SensorImpl sensor) {
239    this.sensor = sensor;
240  }
241
242  /**
243   * @param timestamp the timestamp to set
244   */
245  public void setTimestamp(Date timestamp) {
246    this.timestamp = new Date(timestamp.getTime());
247  }
248
249  /**
250   * @param units the units to set
251   */
252  public void setUnits(String units) {
253    this.units = units;
254  }
255
256
257  /**
258   * @param value the value to set
259   */
260  public void setValue(Double value) {
261    this.value = value;
262  }
263
264  /**
265   * @return The domainmodel Measurement that is equivalent to this
266   *         MeasurementImpl.
267   */
268  public Measurement toMeasurement() {
269    Measurement ret = new Measurement(sensor.getId(), timestamp, value, Unit.valueOf(units));
270    return ret;
271  }
272
273  /* (non-Javadoc)
274   * @see java.lang.Object#toString()
275   */
276  @Override
277  public String toString() {
278    return "MeasurementImpl [pk=" + pk + ", id=" + id + ", sensor=" + sensor + ", timestamp="
279        + timestamp + ", value=" + value + ", units=" + units + ", depository=" + depository + "]";
280  }
