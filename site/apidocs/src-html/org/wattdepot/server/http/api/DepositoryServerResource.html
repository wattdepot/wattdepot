019package org.wattdepot.server.http.api;
020
021import java.util.logging.Level;
022
023import org.restlet.data.Status;
024import org.restlet.resource.ResourceException;
025import org.wattdepot.common.domainmodel.Depository;
026import org.wattdepot.common.domainmodel.Labels;
027import org.wattdepot.common.exception.IdNotFoundException;
028import org.wattdepot.common.exception.MisMatchedOwnerException;
029import org.wattdepot.common.http.api.DepositoryResource;
030
031/**
032 * DepositoryServerResource - Handles Depository HTTP API
033 * ("/wattdepot/{org-id}/depository/{depository-id}").
034 * 
035 * @author Cam Moore
036 * 
037 */
038public class DepositoryServerResource extends WattDepotServerResource implements DepositoryResource {
039
040  /** The depository_id in the request. */
041  private String depositoryId;
042
043  /*
044   * (non-Javadoc)
045   * 
046   * @see org.restlet.resource.Resource#doInit()
047   */
048  @Override
049  protected void doInit() throws ResourceException {
050    super.doInit();
051    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
052  }
053
054  /*
055   * (non-Javadoc)
056   * 
057   * @see org.wattdepot.restlet.DepositoryResource#retrieve()
058   */
059  @Override
060  public Depository retrieve() {
061    getLogger().log(Level.INFO, "GET /wattdepot/{" + orgId + "}/depository/{" + depositoryId + "}");
062    if (isInRole(orgId)) {
063      try {
064        depot.getOrganization(orgId, true);
065      }
066      catch (IdNotFoundException e1) {
067        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, orgId + " is not a defined Organization.");
068      }
069      Depository depo = null;
070      try {
071        depo = depot.getDepository(depositoryId, orgId, true);
072      }
073      catch (IdNotFoundException e) {
074        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Depository " + depositoryId
075            + " is not defined.");
076      }
077      return depo;
078    }
079    else {
080      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
081      return null;
082    }
083  }
084
085  /*
086   * (non-Javadoc)
087   * 
088   * @see org.wattdepot.restlet.DepositoryResource#store(org.wattdepot.
089   * datamodel.Depository)
090   */
091  @Override
092  public void update(Depository depository) {
093    getLogger().log(Level.INFO,
094        "POST /wattdepot/{" + orgId + "}/depository/{" + depositoryId + "} with " + depository);
095    if (isInRole(orgId)) {
096      try {
097        depot.getOrganization(orgId, true);
098      }
099      catch (IdNotFoundException e) {
100        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, orgId + " does not exist.");
101      }
102      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Can't update a Depository.");
103    }
104    else {
105      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
106    }
107  }
108
109  /*
110   * (non-Javadoc)
111   * 
112   * @see org.wattdepot.restlet.DepositoryResource#remove()
113   */
114  @Override
115  public void remove() {
116    getLogger().log(Level.INFO, "DEL /wattdepot/{" + orgId + "}/depository/{" + depositoryId + "}");
117    try {
118      depot.deleteDepository(depositoryId, orgId);
119    }
120    catch (IdNotFoundException e) {
121      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
122    }
123    catch (MisMatchedOwnerException e) {
124      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
125    }
126  }
127
128}
