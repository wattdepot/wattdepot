046/**
047 * OrgVisualizerServerResource - Summary Web page generator for WattDepot
048 * Organizations. It handles ("/wattdepot/{org-id}/visualize/").
049 * 
050 * @author Cam Moore
051 * 
052 */
053public class OrgVisualizerServerResource extends WattDepotServerResource {
054
055  /**
056   * @return The Organization summary information as an HTML Representation.
057   */
058  @Get()
059  public Representation toHtml() {
060    getLogger().log(Level.INFO,
061        "GET " + API.BASE_URI + "{" + orgId + "}/" + Labels.VISUALIZE + "/");
062    if (isInRole(orgId) || isInRole(Organization.ADMIN_GROUP.getId())) {
063      Map<String, Object> dataModel = new HashMap<String, Object>();
064      dataModel.put("orgId", orgId);
065      Representation rep = null;
066      TemplateRepresentation template = null;
067      try {
068        // Long startTime = System.nanoTime();
069        depot.getOrganization(orgId, true);
070        List<Depository> depos = depot.getDepositories(orgId, false);
071        List<Sensor> sensors = depot.getSensors(orgId, false);
072        List<SensorGroup> sensorGroups = depot.getSensorGroups(orgId, false);
073        // Long endTime = System.nanoTime();
074        // Long diff = endTime - startTime;
075        // getLogger().log(Level.INFO,
076        // "getDepositories took " + (diff / 1E9) + " seconds");
077        Map<String, List<Sensor>> depoSensors = new HashMap<String, List<Sensor>>();
078        Map<String, Map<String, List<Date>>> depotSensorInfo = new HashMap<String, Map<String, List<Date>>>();
079        for (Depository d : depos) {
080          Map<String, List<Date>> sensorInfo = new HashMap<String, List<Date>>();
081          depotSensorInfo.put(d.getId(), sensorInfo);
082          List<Sensor> sensorList = new ArrayList<Sensor>();
083          depoSensors.put(d.getId(), sensorList);
084          long startTime = System.nanoTime();
085          for (String sensorId : depot.listSensors(d.getId(), orgId, false)) {
086//            long startTime2 = System.nanoTime();
087            Sensor s = depot.getSensor(sensorId, orgId, false);
088//            long endTime2 = System.nanoTime();
089//            long diff2 = endTime2 - startTime2;
090//            getLogger().log(Level.SEVERE, "getSensor(" + sensorId + ") took " + (diff2 / 1E9) + " seconds");
091            sensorList.add(s);
092            try {
093//              startTime2 = System.nanoTime();
094              InterpolatedValue earliest = depot.getEarliestMeasuredValue(
095                  d.getId(), orgId, sensorId, false);
096              InterpolatedValue latest = depot.getLatestMeasuredValue(
097                  d.getId(), orgId, sensorId, false);
098              List<Date> info = new ArrayList<Date>();
099              info.add(earliest.getStart());
100              info.add(latest.getEnd());
101              sensorInfo.put(sensorId, info);
102//              endTime2 = System.nanoTime();
103//              diff2 = endTime2 - startTime2;
104//              getLogger().log(Level.SEVERE, "Sensor earliest and latest for " + sensorId + " took " + (diff2 / 1E9) + " seconds");
105            }
106            catch (NoMeasurementException e) {
107              // TODO Auto-generated catch block
108              e.printStackTrace();
109            }
110          }
111          long endTime = System.nanoTime();
112          long diff = endTime - startTime;
113          getLogger().log(Level.SEVERE, "Depository " + d.getName() + ": Getting Sensor info took " + (diff / 1E9) + " seconds for " + sensorList.size() + " sensors.");
114          startTime = System.nanoTime();
115          int groupCount = 0;
116          for (String groupId : depot.getSensorGroupIds(orgId, false)) {
117            groupCount++;
118            SensorGroup group = depot.getSensorGroup(groupId, orgId, false);
119            List<Date> info = new ArrayList<Date>();
120//            InterpolatedValue earliest = null;
121            Date earliestDate = null;
122//            InterpolatedValue latest = null;
123            Date latestDate = null;
124            for (String sensorId : group.getSensors()) {
125              List<Date> sensorDates = sensorInfo.get(sensorId);
126              if (sensorDates != null) {
127                Date temp = sensorDates.get(0);
128                if (earliestDate == null) {
129                  earliestDate = temp;
130                }
131                else if (temp.before(earliestDate)) {
132                  earliestDate = temp;
133                }
134                temp = sensorDates.get(1);
135                if (latestDate == null) {
136                  latestDate = temp;
137                }
138                else if (temp.after(latestDate)) {
139                  latestDate = temp;
140                }
141              }
142//              try {
143//                InterpolatedValue temp = depot.getEarliestMeasuredValue(
144//                    d.getId(), orgId, sensorId, false);
145//                if (temp != null) {
146//                  if (earliest == null) {
147//                    earliest = temp;
148//                  }
149//                  else if (earliest.getDate().after(temp.getDate())) {
150//                    earliest = temp;
151//                  }
152//                }
153//                temp = depot.getLatestMeasuredValue(d.getId(), orgId, sensorId,
154//                    false);
155//                if (temp != null) {
156//                  if (latest == null) {
157//                    latest = temp;
158//                  }
159//                  else if (latest.getDate().before(temp.getDate())) {
160//                    latest = temp;
161//                  }
162//                }
163//              }
164//              catch (NoMeasurementException e) {
165//                // TODO Auto-generated catch block
166//                e.printStackTrace();
167//              }
168            }
169            if (earliestDate != null) {
170              info.add(earliestDate);
171            }
172            else {
173              // CAM what is a value we can use for no data?
174              info.add(new Date(0l));
175            }
176            if (latestDate != null) {
177              info.add(latestDate);
178            }
179            else {
180              // CAM what is a value we can use for no data?
181              info.add(new Date(0l));
182            }
183//            if (earliest != null) {
184//              info.add(earliest.getDate());
185//            }
186//            else {
187//              // CAM what is a value we can use for no data?
188//              info.add(new Date(0l));
189//            }
190//            if (latest != null) {
191//              info.add(latest.getDate());
192//            }
193//            else {
194//              // CAM what is a value we can use for no data?
195//              info.add(new Date(0l));
196//            }
197            sensorInfo.put(groupId, info);
198          }
199          endTime = System.nanoTime();
200          diff = endTime - startTime;
201          getLogger().log(Level.SEVERE, "Depository " + d.getName() + ": Getting SensorGroup info took " + (diff / 1E9) + " seconds for " + groupCount + " groups.");
202        }
203        dataModel.put("depositories", depos);
204        dataModel.put("sensors", sensors);
205        dataModel.put("sensorgroups", sensorGroups);
206        dataModel.put("depoSensors", depoSensors);
207        dataModel.put("depotSensorInfo", depotSensorInfo);
208        rep = new ClientResource(LocalReference.createClapReference(getClass()
209            .getPackage()) + "/Visualizer.ftl").get();
210      }
211      catch (IdNotFoundException e) {
212        e.printStackTrace();
213        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
214        return null;
215      }
216      catch (MisMatchedOwnerException e) {
217        e.printStackTrace();
218        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
219        return null;
220      }
221      template = new TemplateRepresentation(rep, dataModel, MediaType.TEXT_HTML);
222      return template;
223    }
224    return null;
225  }
