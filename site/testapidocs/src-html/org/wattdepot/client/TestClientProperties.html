065  /**
066   * Test method for {@link org.wattdepot.client.ClientProperties#get(java.lang.String)}.
067   */
068  @Test
069  public void testGet() {
070    ClientProperties props = new ClientProperties();
071    assertNotNull(props);
072    String result = props.get(ClientProperties.CLIENT_HOME_DIR);
073    assertNotNull(result);
074    String expected = UserHome.getHomeString() + "/.wattdepot3/client";
075    assertEquals(expected, result);
076  }
077
078  /**
079   * Test method for {@link org.wattdepot.client.ClientProperties#setTestProperties()}.
080   */
081  @Test
082  public void testSetTestProperties() {
083    ClientProperties props = new ClientProperties();
084    assertNotNull(props);
085    String port = props.get(ClientProperties.PORT_KEY);
086    props.setTestProperties();
087    String testPort = props.get(ClientProperties.PORT_KEY);
088    assertFalse(port.equals(testPort));
089    String echo = props.echoProperties();
090    assertNotNull(echo);
091  }
