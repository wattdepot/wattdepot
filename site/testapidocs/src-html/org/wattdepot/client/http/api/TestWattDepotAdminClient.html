064  /**
065   * Starts up a WattDepotServer to start the testing.
066   * 
067   * @throws Exception if there is a problem starting the server.
068   */
069  @BeforeClass
070  public static void setupServer() throws Exception {
071    LoggerUtil.disableLogging();    
072    server = WattDepotServer.newTestInstance();
073    try {
074      ClientProperties props = new ClientProperties();
075      props.setTestProperties();
076      logger = Logger.getLogger("org.wattdepot.client");
077      LoggerUtil.setLoggingLevel(logger, props.get(ClientProperties.LOGGING_LEVEL_KEY));
078      LoggerUtil.useConsoleHandler();
079      logger.finest("setUp()");
080      serverURL = "http://" + props.get(ClientProperties.WATTDEPOT_SERVER_HOST) + ":"
081          + props.get(ClientProperties.PORT_KEY) + "/";
082      logger.finest("Using " + serverURL);
083      if (admin == null) {
084        try {
085          admin = new WattDepotAdminClient(serverURL, props.get(ClientProperties.USER_NAME),
086              "admin", props.get(ClientProperties.USER_PASSWORD));
087        }
088        catch (Exception e) {
089          System.out.println("Failed with " + props.get(ClientProperties.USER_NAME) + " and "
090              + props.get(ClientProperties.USER_PASSWORD));
091          e.printStackTrace();
092        }
093      }
094    }
095    catch (Exception e) {
096      e.printStackTrace();
097    }
098  }

110  /**
111   * Test method for Organizations.
112   */
113  @Test
114  public void testOrganizations() {
115    String orgName = "Organization ";
116    OrganizationList list = admin.getOrganizations();
117    int numOrganizations = list.getOrganizations().size();
118    Organization one = new Organization(orgName + 1);
119    admin.putOrganization(one);
120    Organization two = new Organization(orgName + 2);
121    admin.putOrganization(two);
122    Organization three = new Organization(orgName + 3);
123    admin.putOrganization(three);
124    list = admin.getOrganizations();
125    assertNotNull(list);
126    assertTrue("Expecting " + (numOrganizations + 3) + " got " + list.getOrganizations().size(),
127        (numOrganizations + 3) == list.getOrganizations().size());
128    try {
129      admin.deleteOrganization(three.getId());
130    }
131    catch (IdNotFoundException e) {
132      e.printStackTrace();
133      fail(e.getMessage() + " should delete defined organization " + three.getName());
134    }
135    try {
136      admin.deleteOrganization("bogus-org");
137      fail("Should not be able to delete a bogus organization.");
138    }
139    catch (IdNotFoundException e) {
140      // expected
141    }
142    list = admin.getOrganizations();
143    assertNotNull(list);
144    assertTrue(numOrganizations + 2 == list.getOrganizations().size());
145    two.setName("New organization name");
146    admin.updateOrganization(two);
147    try {
148      Organization defined = admin.getOrganization(two.getId());
149      assertNotNull(defined);
150      assertTrue(defined.equals(two));
151    }
152    catch (IdNotFoundException e) {
153      e.printStackTrace();
154      fail(e.getMessage() + " should be able to update " + two.getId());
155    }
156
157    // Clean up
158    if (admin.isDefinedOrganization(one.getId())) {
159      try {
160        admin.deleteOrganization(one.getId());
161      }
162      catch (IdNotFoundException e) {
163        e.printStackTrace();
164      }
165    }
166    if (admin.isDefinedOrganization(two.getId())) {
167      try {
168        admin.deleteOrganization(two.getId());
169      }
170      catch (IdNotFoundException e) {
171        e.printStackTrace();
172      }
173    }
174    if (admin.isDefinedOrganization(three.getId())) {
175      try {
176        admin.deleteOrganization(three.getId());
177      }
178      catch (IdNotFoundException e) {
179        e.printStackTrace();
180      }
181    }
182  }

184  /**
185   * Test for manipulating Users.
186   */
187  @Test
188  public void testUsers() {
189    String orgName = "Group ";
190    Organization one = new Organization(orgName + 1);
191    admin.putOrganization(one);
192    Organization two = new Organization(orgName + 2);
193    admin.putOrganization(two);
194    // create two users with same id different organizations.
195    UserInfo u1 = new UserInfo("id", "firstName", "lastName", "email", one.getId(),
196        new HashSet<Property>(), "secret1");
197    admin.putUser(u1);
198    UserInfo u2 = new UserInfo("id", "firstName", "lastName", "email", two.getId(),
199        new HashSet<Property>(), "secret2");
200    admin.putUser(u2);
201    UserInfoList ulist = admin.getUsers(one.getId());
202    assertNotNull(ulist);
203    assertTrue("Expecting 1 got " + ulist.getUsers().size(), ulist.getUsers().size() == 1);
204    ulist = admin.getUsers(two.getId());
205    assertNotNull(ulist);
206    assertTrue(ulist.getUsers().size() == 1);
207    u1.setEmail("new email");
208    admin.updateUser(u1);
209    ulist = admin.getUsers(one.getId());
210    assertNotNull(ulist);
211    assertTrue("Expecting 1 got " + ulist.getUsers().size(), ulist.getUsers().size() == 1);
212    assertTrue("didn't change email", ulist.getUsers().get(0).getEmail().equals("new email"));
213    // try to delete undefined user
214    try {
215      admin.deleteUser("bogus-id", u1.getOrganizationId());
216      fail("Should not be able to delete a bogus user.");
217    }
218    catch (IdNotFoundException e) {
219      // this is expected
220    }
221    assertFalse("isDefinedUser('bogus') should return false.", admin.isDefinedUserInfo("bogus-id", u1.getOrganizationId()));
222    // passwords
223    try {
224      UserPassword pass1 = admin.getUserPassword(u1.getUid(), u1.getOrganizationId());
225      assertNotNull(pass1);
226      pass1.setEncryptedPassword(StrongAES.getInstance().encrypt("foooo"));
227      try {
228        admin.updateUserPassword(pass1);
229      }
230      catch (IdNotFoundException e) {
231        e.printStackTrace();
232        fail("problem updating password");
233      }
234    }
235    catch (IdNotFoundException e) {
236      fail("u1 is defined");
237    }
238
239    // Clean up
240    if (admin.isDefinedUserInfo(u1.getUid(), u1.getOrganizationId())) {
241      try {
242        admin.deleteUser(u1.getUid(), u1.getOrganizationId());
243      }
244      catch (IdNotFoundException e) {
245        e.printStackTrace();
246      }
247    }
248    if (admin.isDefinedUserInfo(u2.getUid(), u2.getOrganizationId())) {
249      try {
250        admin.deleteUser(u2.getUid(), u2.getOrganizationId());
251      }
252      catch (IdNotFoundException e) {
253        e.printStackTrace();
254      }
255    }
256    if (admin.isDefinedOrganization(one.getId())) {
257      try {
258        admin.deleteOrganization(one.getId());
259      }
260      catch (IdNotFoundException e) {
261        e.printStackTrace();
262      }
263    }
264    if (admin.isDefinedOrganization(two.getId())) {
265      try {
266        admin.deleteOrganization(two.getId());
267      }
268      catch (IdNotFoundException e) {
269        e.printStackTrace();
270      }
271    }

273  }
